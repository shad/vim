### Install ###

    ## CREATE .vim dir and link vimrc
    cd ~
    git clone git://github.com/shad/vim.git .vim
    ln -s ~/.vim/vimrc ~/.vimrc

    # INSTALL VUNDLE
    cd .vim
    git clone https://github.com/VundleVim/Vundle.vim.git ~/.vim/bundle/Vundle.vim
